# Async image compression

    nano gulpfile

    #  Change the variables: 
    #  SOURCE='your/source/path'
    #  DESCTINATION='your/target/path' 

    npm install 

    npm update

    cd source
    # copy images you want to compress to the source folder

    cp -R /source/path/to/your/images/ . 

    cd ..

    npm run compress
